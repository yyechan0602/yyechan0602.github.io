    public static void House(String num){
        int[] memory = new int[10];
        for ( int i=0; i<10; i++) {
            memory[i] = 0;
        }

        String[] nums = num.split("");
        for ( String i : nums) {
            memory[Integer.parseInt(i)] += 1;
        }
        memory[6] = ((memory[6] + memory[9]) +1 )/2;

        int result = 0;
        for ( int i=0; i<9; i++) {
            if(result < memory[i]){
                result = memory[i];
            }
        }
        System.out.println(result);
    }
